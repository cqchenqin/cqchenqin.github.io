C语言复习篇（五）

实验五 利用数组处理批量数据程序设计

1.输入 10 个整数，用选择法对这 10 个整数按从小到大的顺序排序并输出排序后的结果。

1.Enter 10 integers, sort them from small to large, and output the sorted results by Selection sort method.

#include <stdio.h>
int main ()
{
    int i=0,j=0,n=0;
    printf ("请输入所需要排序的总数：n=");
    scanf ("%d",&n); 
    float a[n],t=0;
    for (i=0;i<n;i++)
    {
        scanf ("%f",&a[i]);
    }
    for (i=0;i<n-1;i++)
    {
        for (j=0;j<n-i-1;j++)
        {
            if (a[j]>a[j+1]) t=a[j],a[j]=a[j+1],a[j+1]=t;
        }
    }
    printf ("这%d个数从小到大排序顺序为：",n);
    for (i=0;i<n;i++)
    {
        if (i<n-1) printf ("%.2f < ",a[i]);
        else printf ("%.2f",a[i]);
    }
 }

2.青年歌手参加歌曲大奖赛，有 10 个评委进行打分，编程求这位选手的平均得分（去掉一个最高分和一个最低分）。

#include <stdio.h>
int main ()
{
    int n=0,i=0;
    printf ("请输入评委老师总数：n=");
    scanf ("%d",&n);
    float a[n],max=0,min=0,sum=0;
    printf ("请分别输入 %d 个评委老师所给分数：",n);
    for (i=0;i<n;i++)
    {
        scanf ("%f",&a[i]);
    }
    for (max=a[0],min=a[0],sum=a[0],i=1;i<n;i++)
    {
        sum=sum+a[i];
        if (a[i]>max) max=a[i];
        else if (a[i]<min) min=a[i];
    }
    sum=(sum-max-min)/(n-2);
    printf ("这位选手的平均得分为：average=%.2f",sum);
 }

3.从键盘输入一整数，然后在一个整型一维数组 a[20]中(数据按从小到大顺序)，用折半查找法找出该数是数组中第几个元素的值。如果该数不在数组中，则打印”No found”。

#include <stdio.h>
int main ()
{
    int n=0,i=0,x=0,b=0,c=0,d=0,t=-1;
    printf ("请输入这些整数的数量：n=");
    scanf ("%d",&n);
    int a[n];
    printf ("请输入 %d 个整数（按照从小到大的顺序）：",n);
    for (i=0;i<n;i++)
    {
        scanf ("%d",&a[i]);
    }
    printf ("请输入需要查找位置的整数：x=");
    scanf ("%d",&x);
    for (c=0,d=n-1,b=(c+d)/2;b!=c;b=(c+d)/2)
    {
        if (a[b]==x) 
        {
            t=b;break;
        }
        else if (a[b]<x) c=b;
        else d=b;
    }
    if (t==-1) printf ("No found");
    else printf ("%d是数组中第%d个元素的值",x,t+1);
 }

4.为整型二维数组 b[3][4]输入 12 个数据，计算并输出数组中所有正数之和、所有负数之和。

#include <stdio.h>
int main ()
{
    int b[3][4],i=0,j=0,x=0,y=0;
    printf ("请输入12个整数：");
    for (i=0;i<3;i++)
    {
        for (j=0;j<4;j++)
        {
            scanf ("%d",&b[i][j]);
        }
    }
    for (i=0;i<3;i++)
    {
        for (j=0;j<4;j++)
        {
            if (b[i][j]>0) x=x+b[i][j];
            else if (b[i][j]<0) y=y+b[i][j];
        }
    }
    printf ("数组中所有正数之和为：%d；\n所有负数之和为：%d。",x,y);
 }

5.请找出二维数组 b[5][4]中的鞍点，鞍点指该位置上的元素值在该行中最大，且在该列中最小。如果有鞍点，请输出其下标值；如果没有鞍点，请给出没有鞍点的提示信息。

#include <stdio.h>
int main ()
{
    int b[5][4],i=0,j=0,k=0,max=0,min=0,t=-1,x=0,y=-1;
    printf ("请输入二维数组 b[5][4]数据（没行五个数字，共四行）：");
    for (i=0;i<4;i++)
    {
        printf ("\n第 %d 行：",i+1);
        for (j=0;j<5;j++)
        {
            scanf ("%d",&b[j][i]);
        }
    }
    for (i=0;i<4;i++,x=0)
    {
        for (t=0,j=0,max=b[j][i];j<5;j++)
        {
            if (b[j][i]>max) max=b[j][i],t=j;
        }
        for (k=0,min=b[t][i];k<4;k++)
        {
            if (b[t][k]<min)
            {
                x=-1;break;
            }
        }
        if (x==0) y=0,printf ("鞍点为b[%d][%d] (注：其中b[x][y],x<5,y<4)",t,i);
    }
    if (y==-1) printf ("没有鞍点！！！");
 }

7.有一电文，已按下列规律译成译码：

A→Z　　a→z

B→Y　　b→y

C→X　　c→x

即第一个字母变成第 26 个字母，第 i 个字母变成第(26-i+1)个字母。非字母字符不变。编写一个程序将密码译成原文，并输出密码和原文。

#include <stdio.h>
#include <string.h>
int main ()
{
    char a[120],b[120];
    int i=0;
    printf ("请输入需要加密文本：");
    gets(a);
    for (i=0;i<strlen(a);i++)
    {
        if ('a'<=a[i]&&a[i]<='z') b[i]='a'+'z'-a[i];
        else if ('A'<=a[i]&&a[i]<='Z') b[i]='A'+'Z'-a[i];
        else b[i]=a[i];
    }
    printf ("原文为：");
    puts (a);
    printf ("\n加密结果为：");
    puts (b);
}

*8. （选做题）有一个已排序好的数组，今输入一个数，要求按原来排序的规律将它插入在数组中。

#include <stdio.h>
int main ()
{
    int a[20],i=0,t=0,n=0,x=0;
    printf ("请输入原数组数字个数：n=");
    scanf ("%d",&n);
    printf ("请输入原数组数：\n");
    for (i=0;i<n;i++)
    {
        scanf ("%d",&a[i]);
    }
    printf ("请输入所需要插入的数字：x=");
    scanf ("%d",&x); 
    for (i=0,t=0;i<n;i++)
    {
        if (a[i]<x) t++;
        else ;
    }
    printf ("插入以后的数组为：\n");
    for (i=0;i<=n;i++)
    {
        if (i<t) printf (" %d ",a[i]);
        else if (i==t) printf (" %d ",x);
        else printf (" %d ",a[i-1]);
    }
}

*9. （选做题）用筛选法求出 100 之内的所有素数。

#include <stdio.h>
int main ()
{
    int a[101],i=0,j=0,t=0;
    for (i=1;i<101;i++)
    {
        a[i]=i;
    }
    for (a[1]=0,i=2;i<101;i++,t=0)
    {
        if (i==2) t=-1;
        else 
        {
            for (j=2;j<i;j++)
            {
                if (a[i]%j==0) t=-1;
            }
        }
        if (t==-1) 
        {
            a[i]=0;
            for (j=2;i*j<101;j++)
            {
                a[i*j]=0;
            }
        }
    }
    printf ("100 之内的所有素数:\n");
    for (i=1;i<101;i++)
    {
        if (a[i]!=0) printf (" %d ",i);
    }
 }

#include <stdio.h>
#include <string.h>

*12. （选做题）有一篇文章，共 3 行文字，每行 80 个字符。要求分别统计出英文大写字母、小写字母、数字、空格，以及其他字符的个数。(提示：可用二维字符数组（3 行 80 列）存放该文章。)

#include <stdio.h>
int main ()
{
    char x[80][3];
    int i=0,j=0,a=0,b=0,c=0,d=0,e=0;
    for (j=0;j<3;j++)
    {
        for (i=0;i<80;i++)
        {
            scanf ("%c",&x[i][j]);
        }
    }
    for (j=0;j<3;j++)
    {
        for (i=0;i<80;i++)
        {
            if ('A'<=x[i][j]&&x[i][j]<='Z') a++;
            else if ('a'<=x[i][j]&&x[i][j]<='z') b++;
            else if ('0'<=x[i][j]&&x[i][j]<='9') c++;
            else if (x[i][j]==' ') d++;
            else e++;
        }
    }
    printf ("英文大写字母：%d 个;\n小写字母：%d 个;\n数字：%d 个;\n空格：%d 个;\n其他字符：%d 个。",a,b,c,d,e);
}
